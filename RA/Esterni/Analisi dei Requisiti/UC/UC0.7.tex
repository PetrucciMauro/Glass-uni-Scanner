	\textbf{Attori}: utente autenticato\ped{g}. \\
	\textbf{Descrizione}: l'utente può modificare i contenuti (File\ped{g} media, presentazioni e Infografiche\ped{g}) presenti nel proprio Account\ped{g}, inserendone di nuovi, rinominando o eliminando quelli presenti. \\
	\textbf{Precondizione}: il sistema è attivo e funzionante; l'utente ha effettuato il Login\ped{g} e desidera effettuare modifiche ai propri File\ped{g} nel proprio Account\ped{g}.	\\
	\textbf{Postcondizione}: l'utente ha eseguito l'operazione voluta sul contenuto che desiderava: ha inserito un nuovo File\ped{g} media nel proprio spazio oppure ha rinominato una presentazione o una Infografica\ped{g} altrimenti eliminato una presentazione, Infografica\ped{g} o File\ped{g} media.	\\
	\textbf{Scenario principale}:
	\begin{enumerate}
		\item Inserimento File\ped{g} media \S\hyperref[uc0.7.1]{(UC 0.7.1)};
		\item Eliminazione File\ped{g} media \S\hyperref[uc0.7.2]{(UC 0.7.2)};
		\item Rinomina presentazione \S\hyperref[uc0.7.3]{(UC 0.7.3)};
		\item Rinomina Infografica\ped{g} \S\hyperref[uc0.7.4]{(UC 0.7.4)};
		\item Rinomina File\ped{g} media \S\hyperref[uc0.7.7]{(UC 0.7.7)};
		\item Eliminazione presentazioni \S\hyperref[uc0.7.5]{(UC 0.7.5)};
		\item Eliminazione Infografica\ped{g} \S\hyperref[uc0.7.6]{(UC 0.7.6)};
		\item Rinomina File\ped{g} media \S\hyperref[uc0.7.7]{(UC 0.7.7)};
		\item Visualizzazione delle proprie presentazioni \S\hyperref[uc0.7.8]{(UC 0.7.8)};
		\item Visualizzazione delle proprie Infografiche\ped{g} \S\hyperref[uc0.7.9]{(UC 0.7.9)};
		\item Visualizzazione dei propri File\ped{g} media \S\hyperref[uc0.7.10]{(UC 0.7.10)};
	\end{enumerate}
	}
\subsubsection{UC 0.7.1 - Inserimento File media}{
	\label{uc0.7.1}
	\textbf{Attori}: utente autenticato\ped{g}. \\
	\textbf{Descrizione}: l'utente ha la possibilità di caricare un File\ped{g} media (audio, video, immagine) contenuto nel proprio computer o dispositivo. \\
	\textbf{Precondizione}: il sistema ha caricato lo spazio dedicato ai File\ped{g} media dell'utente e l'utente desidera caricare sul Server\ped{g} un File\ped{g} media esistente in locale.	\\
	\textbf{Postcondizione}: l'utente ha caricato sul Server\ped{g} i File\ped{g} media selezionati.	\\
	\textbf{Scenario principale}:
	\begin{enumerate}
		\item Navigazione nello spazio di lavoro;
		\item Selezione File\ped{g} media;
		\item Conferma e caricamento.
	\end{enumerate}
	\textbf{Scenari alternativi}: 
	\begin{itemize}
		\item L'operazione viene annullata e non si apportano cambiamenti.
	\end{itemize}
	}
