\subsection{UC 1 - Gestione lavoro}{
	\label{uc1}
	\begin{figure}[H]
		\centering
		\includegraphics[scale=0.6]{\imgs {UC1}.jpg} %inserire il diagramma UML
		\label{fig:uc1}
		\caption{Caso d'uso 1: Gestione lavoro}
	\end{figure}
	\textbf{Attori}: utente, utente desktop, utente mobile. \\
	\textbf{Descrizione}: il sistema deve mettere l'utente in condizioni di raccogliere idee e materiale multimediale da essere utilizzati in una presentazione, creare e modificare una presentazione, creare infografiche a partire da una presentazione, salvare ciò che ha prodotto su di un server ed infine eseguire una presentazione creata con il sistema. \\
	\textbf{Precondizione}: il sistema è funzionante e pronto all'utilizzo, l'utente ha effettuato il login ed è entrato nel sito nella parte di gestione del proprio lavoro.	\\
	\textbf{Postcondizione}: è stato eseguito ciò che l'utente desiderava per gestire il proprio lavoro.	\\
	\textbf{Scenario principale}:
	\begin{enumerate}
		\item l'utente desktop può creare una nuova presentazione \hyperref[uc1.2]{(UC 1.2)};
		\item l'utente autenticato può modificare una presentazione da desktop \hyperref[uc1.3]{(UC 1.3)} e da mobile \hyperref[uc1.4]{(UC 1.4)};
		\item l'utente può eseguire la presentazione \hyperref[uc1.18]{(UC 1.18)};
		\item l'utente desktop può creare un'infografica della presentazione \hyperref[uc1.6]{(UC 1.6)};
		\item l'utente desktop può modificare un'infografica \hyperref[uc1.17]{(UC 1.17)}.
	\end{enumerate}
	}
