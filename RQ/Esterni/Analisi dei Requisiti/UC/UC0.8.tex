\subsection{UC 0.8 - Registrazione account}{
	\label{uc0.8}
	\begin{figure}[H]
		\centering
		\includegraphics[scale=0.75]{\imgs {UC0.8}.jpg}
		\label{fig:uc0.8}
		\caption{Caso d'uso 0.8: Registrazione nuovo account}
	\end{figure}
	\textbf{Attori}: utente non autenticato\ped{g}. \\
	\textbf{Descrizione}: l'utente non autenticato\ped{g} desidera registrarsi al sistema, inserisce uno username alfanumerico e password alfanumerica di almeno sei caratteri. Dopo aver ricevuto conferma dal sistema l'utente viene registrato. \\
	\textbf{Precondizione}: il sistema è attivo e funzionante; l'utente non autenticato\ped{g} chiede d'effettuare la registrazione.	\\
	\textbf{Postcondizione}: la registrazione è stata effettuata, il sistema è al punto iniziale.	\\
	\textbf{Scenario principale}:
	\begin{enumerate}
		\item Inserimento username per la registrazione \S\hyperref[uc0.8.1]{(UC 0.8.1)};
		\item Inserimento password per la registrazione \S\hyperref[uc0.8.2]{(UC 0.8.2)};
		\item Reinserimento password per la registrazione \S\hyperref[uc0.8.3]{(UC 0.8.3)};
		\item L'utente viene registrato nel sistema.
	\end{enumerate}
	\textbf{Scenari alternativi}: 
	\begin{itemize}
		\item Lo username scelto risulta essere appartenente ad un altro utente; l'utente non ancora autenticato che si sta registrando viene avvisato dell'errore e può sceglierne un altro;
		\item La password e la sua conferma risultano essere diverse; l'utente non ancora autenticato viene avvisato dell'errore e può reinserire i dati;
		\item La password risulta non conforme ai vincoli adottati dal sistema; l'utente non ancora autenticato viene avvisato dell'errore e può reinserire i dati.
	\end{itemize}
	}
	\subsubsection{UC0.8.1 - Inserimento username per la registrazione}{
		\label{uc0.8.1}
		\textbf{Attori}: utente non autenticato\ped{g}. \\
		\textbf{Descrizione}: l'utente non autenticato\ped{g} inserisce il proprio username.	\\
		\textbf{Precondizione}: l'utente non autenticato\ped{g} è pronto ad inserire lo username scelto.	\\
		\textbf{Postcondizione}: è stato inserito lo username.\\
		\textbf{Scenario principale}:
		\begin{enumerate}
			\item L'utente inserisce il proprio username.
		\end{enumerate}
		\textbf{Scenario alternativo}:
		\begin{enumerate}
			\item Se lo username  inserito è già in uso nel sistema l'utente visualizza un messaggio di errore (UC0.8.4)
		\end{enumerate}
		}
	\subsubsection{UC0.8.2 - Inserimento password per la registrazione} {
		\label{uc0.8.2}
		\textbf{Attori}: utente non autenticato\ped{g}. \\
		\textbf{Descrizione}: l'utente non autenticato\ped{g} inserisce la propria password.	\\
		\textbf{Precondizione}: l'utente non autenticato\ped{g} chiede d'effettuare la registrazione è pronto ad inserire la password scelta.	\\
		\textbf{Postcondizione}: è stata inserita la password.\\
		\textbf{Scenario principale}:
		\begin{enumerate}
			\item L'utente inserisce la propria password.
		\end{enumerate}
		}
	\subsubsection{UC0.8.3 - Reinserimento password per la registrazione} {
		\label{uc0.8.3}
		\textbf{Attori}: utente non autenticato\ped{g}. \\
		\textbf{Descrizione}: l'utente non autenticato\ped{g} reinserisce la propria password.	\\
		\textbf{Precondizione}: l'utente non autenticato\ped{g} richiede di effettuare la registrazione ed è pronto a reinserire la password di conferma.	\\
		\textbf{Postcondizione}: è stata inserita la conferma della password.\\
		\textbf{Scenario principale}:
		\begin{enumerate}
			\item L'utente reinserisce la propria password.
		\end{enumerate}
		}
		\textbf{Scenario alternativo}:
		\begin{enumerate}
			\item Se la password e la sua conferma sono diverse l'utente visualizza un messaggio di errore(UC0.8.5)
		\end{enumerate}
	\subsubsection{UC0.8.4 - Visualizza errore utente già in uso} {
		\label{uc0.8.4}
		\textbf{Attori}: utente non autenticato\ped{g}. \\
		\textbf{Descrizione}: l?utente non autenticato\ped{g} ha inserito uno username già registrato e richiesto la registrazione	\\
		\textbf{Precondizione}:  l?utente non autenticato\ped{g} ha inserito username, password e la sua conferma e richiesto la registrazione	\\
		\textbf{Postcondizione}: l?utente non autenticato\ped{g} ha visualizzato l?errore e può reinserire i dati per la registrazione  \\
		\textbf{Scenario principale}:
		\begin{enumerate}
			\item Viene visualizzato un messaggio di errore in cui viene detto che  lo username inserito dall?utente è già stato registrato
		\end{enumerate}
		}
	\subsubsection{UC0.8.5 - Visualizza errore password inserite diverse} {
		\label{uc0.8.5}
		\textbf{Attori}: utente non autenticato\ped{g}. \\
		\textbf{Descrizione}: l?utente non autenticato\ped{g} ha inserito valori diversi in password e nella sua conferma	\\
		\textbf{Precondizione}: l?utente non autenticato\ped{g} ha inserito username, password e la sua conferma e richiesto la registrazione	\\
		\textbf{Postcondizione}: l?utente non autenticato\ped{g} ha visualizzato l?errore e può reinserire i dati per la registrazione  \\
		\textbf{Scenario principale}:
		\begin{enumerate}
			\item Viene visualizzato un messaggio di errore in cui viene detto che  la password inserita non è la stessa inserita nella sua conferma
		\end{enumerate}
		}
