\subsection{UC 1.6 - Creazione di un'infografica}{
	\label{uc1.6}
	\textbf{Attori}: utente Desktop\ped{g}. \\
	\textbf{Descrizione}: l'utente è in grado di prendere gli elementi\ped{g} di una presentazione e di inserirli in un unico documento stampabile. \\
	\textbf{Precondizione}: il sistema è attivo e funzionante.	\\
	\textbf{Postcondizione}: l'utente ha creato un'Infografica\ped{g}.	\\
	\textbf{Scenario principale}:
	\begin{enumerate}
		\item Selezione dell'opzione di creazione di una nuova Infografica\ped{g};
		\item Assegnazione del nome all'Infografica\ped{g} creata;
		\item Una volta creata l'Infografica\ped{g} si entra sempre in modalità modifica Infografica\ped{g} \S\hyperref[uc1.17]{(UC 1.17)}.
	\end{enumerate}
	}
