\subsection{UC 0.16 - Download presentazione}{
	\label{uc0.16}
	\textbf{Attori}: utente autenticato\ped{g}.	\\
	\textbf{Descrizione}: l'utente ha la possibilità di salvare una presentazione contenuta in una cartella all'interno del proprio spazio di lavoro sul Server\ped{g}. \\
	\textbf{Precondizione}: il sistema è attivo e funzionante; l'utente ha effettuato il Login\ped{g} e desidera scaricare nel proprio dispositivo una presentazione tra quelle presenti nel suo spazio sul Server\ped{g}.	\\
	\textbf{Postcondizione}: l'utente ha scaricato la presentazione che desiderava.	\\
	\textbf{Scenario principale}:
	\begin{enumerate}
		\item Navigazione nel proprio spazio di lavoro sul Server\ped{g};
		\item Selezione della presentazione tra quelle presenti;
		\item Conferma e scaricamento della presentazione selezionata.
	\end{enumerate}
	\textbf{Scenari alternativi}: 
	\begin{itemize}
		\item L'operazione viene annullata.
	\end{itemize}
	}
