\subsection{UC0.13 - Logout}{
	\label{uc0.13}
	\textbf{Attori}: utente autenticato\ped{g}.\\
	\textbf{Descrizione}: l'utente autenticato\ped{g} effettua il Logout\ped{g} dal suo profilo diventando utente non autenticato\ped{g}.\\
	\textbf{Precondizione}: l'utente è autenticato al sistema.\\
	\textbf{Postcondizione}: l'utente non è più autenticato al sistema.\\
	\textbf{Scenario principale}:
	\begin{enumerate}
		\item Selezione azione di Logout\ped{g};
		\item Conferma Logout\ped{g};
		\item L'utente non è più autenticato al sistema.
	\end{enumerate}
}
