\subsection{UC 0 - Gestione account}{
	\label{uc0}
	\begin{figure}[H]
		\centering
		\includegraphics[scale=0.5]{\imgs {UC0}.jpg} %inserire il diagramma UML
		\label{fig:uc0}
		\caption{Caso d'uso 0: Gestione account}
	\end{figure}
	\textbf{Attori}: utente, utente non autenticato\ped{g}, utente autenticato\ped{g}, amministratore. \\
	\textbf{Descrizione}: il sistema deve permettere all'utente di registrare il proprio Account\ped{g}, accedervi e gestirlo. \\
	\textbf{Precondizione}: il sistema è funzionante.	\\
	\textbf{Postcondizione}: l'utente ha eseguito le operazioni desiderate per gestire il proprio Account\ped{g}.	\\
	\textbf{Scenario principale}:
	\begin{enumerate}
		\item l'utente si registra creando un nuovo Account\ped{g} utente \S\hyperref[uc0.8]{(UC 0.8)};
		\item l'utente non autenticato\ped{g} può effettuare il Login\ped{g} \S\hyperref[uc0.10]{(UC 0.10)} e il Logout\ped{g} \S\hyperref[uc0.13]{(UC 0.13)};
		\item l'utente autenticato\ped{g} può gestire il proprio Account\ped{g} \S\hyperref[uc0.11]{(UC 0.11)};
		\item l'utente può eliminare le presentazioni salvate in locale \S\hyperref[uc0.15]{(UC 0.15)};
		\item l'utente autenticato\ped{g} può gestire il proprio archivio sul Server\ped{g} \S\hyperref[uc0.7]{(UC 0.7)};
		\item l'utente autenticato\ped{g} può scaricare in locale una presentazione dal Server\ped{g} \S\hyperref[uc0.16]{(UC 0.16)};
		\item l'utente autenticato\ped{g} può scaricare un'Infografica\ped{g} nel proprio dispositivo \S\hyperref[uc0.12]{(UC 0.12)};
		\item l'amministratore può gestire i Template\ped{g} delle Infografiche\ped{g} \S\hyperref[uc0.9]{(UC 0.9)}.
	\end{enumerate}
	}
